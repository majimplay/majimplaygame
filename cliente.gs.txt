function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
        const ss = SpreadsheetApp.getActiveSpreadsheet();
    //var spreadsheetId = '1bgW-FdmwgbIUN6nES8Q2alyYVMnxApVwwshxSG0ZjHI'; // Substitua pelo ID da sua planilha
    var sheet = SpreadsheetApp.openById(spreadsheetId).getSheetByName('Clientes');
    var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
    var nextRow = sheet.getLastRow() + 1;
    // Supondo que os dados incluem campos do formulário e dados do usuário
    var newData = {
      'Cliente': data.cliente,
      'CPF': data.cpf,
      'Telefone': data.tel,
      'CEP': data.cep,
      'Logradouro': data.logradouro,
      'Cidade': data.cidade,
      'Estado': data.estado,
      'Bairro': data.bairro,
      'Timestamp': new Date(),
      'Nome': data.name, // Dados do Google Sign-In
      'E-mail': data.email,
      'GoogleID': data.sub
    };
    for (var key in newData) {
      var colIndex = headers.indexOf(key);
      if (colIndex !== -1) {
        sheet.getRange(nextRow, colIndex + 1).setValue(newData[key]);
      } else {
        Logger.log('Coluna não encontrada: ' + key);
      }
    }
    return ContentService.createTextOutput(JSON.stringify({ status: 'success' })).setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    Logger.log(error.toString());
    return ContentService.createTextOutput(JSON.stringify({ status: 'error', message: error.toString() })).setMimeType(ContentService.MimeType.JSON);
  }
}
